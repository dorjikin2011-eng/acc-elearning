async function getLearnerProgress(id) {

  const { data, error } = await supabase.from('progress').select('*').eq('learner_id', id).maybeSingle();

  if (error) console.error(error);

  if (data) {

    return {

      currentMod: data.current_mod,

      currentSlide: data.current_slide,

      quizScores: data.quiz_scores || {},

      completedMods: data.completed_mods || [],

      pct: data.pct || 0,

      completedAt: data.completed_at,

      lastSeen: data.last_seen

    };

  }

  return { currentMod: 0, currentSlide: 0, completedMods: [], quizScores: {}, completedAt: null, lastSeen: null, pct: 0 };

}


async function setLearnerProgress(id, prog) {

  const payload = {

    learner_id: id,

    current_mod: prog.currentMod,

    current_slide: prog.currentSlide,

    quiz_scores: prog.quizScores,

    completed_mods: prog.completedMods,

    pct: prog.pct || 0,

    completed_at: prog.completedAt,

    last_seen: new Date().toISOString()

  };

  const { error } = await supabase.from('progress').upsert(payload, { onConflict: 'learner_id' });

  if (error) console.error("Data syncing issue:", error);

}

        phone=v('reg-phone'),pin=v('reg-pin');

      feGen.textContent = 'This email is already registered. Please sign in instead.';

    } else {

      feGen.textContent = 'Network communication failed. Please verify setup constraints.';

    }

    feGen.style.display = 'block'; 

    return;

  }


  feGen.style.display='none';

  currentLearner = payload;

  showToast('Registration successful! Welcome, '+fname+' 🎉');

  await startCourse();

}

}

let completedMods = new Set();


async function startCourse(){

  const prog = await getLearnerProgress(currentLearner.id);

  currentMod = prog.currentMod || 0;

  currentSlide = prog.currentSlide || 0;

  quizState = prog.quizScores || {};

  completedMods = new Set(prog.completedMods || []);

  

  document.getElementById('learner-greeting').textContent =

    currentLearner.fname+' '+currentLearner.lname+' · '+currentLearner.org+' · '+currentLearner.role;

  

  await showScreen('screen-course');

  renderTabs(); 

  renderSlide(); 

  await updateProgress();

}


async function saveCourseProgress(){

  if(!currentLearner) return;

  const totalSlides = getTotalSlides();

  const doneSlides = getDoneSlides();

  const calculatedPct = Math.round((doneSlides / totalSlides) * 100);


  const prog = {

    currentMod, 

    currentSlide,

    quizScores: quizState, 

    pct: calculatedPct,

    completedMods: Array.from(completedMods),

    completedAt: (currentMod === MODULES.length - 1 && currentSlide === MODULES[currentMod].slides.length - 1) ? new Date().toISOString() : null

  };

  await setLearnerProgress(currentLearner.id, prog);

}


function getTotalSlides(){return MODULES.reduce((a,m)=>a+m.slides.length,0);}

function getDoneSlides(){

  return MODULES.slice(0,currentMod).reduce((a,m)=>a+m.slides.length,0)+currentSlide;

}


async function updateProgress(){

  const pct = Math.round((getDoneSlides()/getTotalSlides())*100);

  document.getElementById('prog').style.width=pct+'%';

  document.getElementById('prog-label').textContent=pct+'% complete · Progress auto-saved';

  await saveCourseProgress();

}


function renderTabs(){

  document.getElementById('nav-tabs').innerHTML=MODULES.map((m,i)=>`

    <button class="tab-btn ${i===currentMod?'active':''} ${completedMods.has(i)?'done':''} ${i>currentMod&&!completedMods.has(i)?'locked':''}"

      onclick="${i<=currentMod||completedMods.has(i)?`goMod(${i})`:'void(0)'}">${m.title}</button>

  `).join('');

}


async function goMod(i){

  if(i>currentMod && !completedMods.has(i)) return;

  currentMod=i; currentSlide=0; quizState={};

  renderTabs(); renderSlide(); await updateProgress();

}


async function nextSlide(){

  const mod=MODULES[currentMod];

  if(currentSlide<mod.slides.length-1){currentSlide++; renderSlide(); await updateProgress();}

}


async function prevSlide(){

  if(currentSlide>0){currentSlide--; renderSlide(); await updateProgress();}

}


async function nextMod(){

  completedMods.add(currentMod);

  currentMod++; currentSlide=0; quizState={};

  renderTabs(); renderSlide(); await updateProgress();

  showToast('Module '+(currentMod)+' unlocked!');

}

function initDrag(){

  document.querySelectorAll('[draggable="true"]').forEach(el=>{

    el.addEventListener('dragstart',e=>e.dataTransfer.setData('text',e.target.id));

  });

  document.querySelectorAll('.drag-target').forEach(t=>{

    t.addEventListener('dragover',e=>{e.preventDefault();t.classList.add('over');});

    t.addEventListener('dragleave',()=>t.classList.remove('over'));

    t.addEventListener('drop',e=>{

      e.preventDefault(); t.classList.remove('over');

      const el=document.getElementById(e.dataTransfer.getData('text'));

      if(el) t.querySelector('.drag-items').appendChild(el);

    });

  });

}


function checkDrag(id,answers){

  let ok=0;

  answers.forEach((ans,i)=>{

    document.getElementById('dt_'+id+'_'+i).querySelectorAll('.drag-item').forEach(el=>{

      if(parseInt(el.dataset.idx)===ans){el.style.background='#4caf50';ok++;}

      else el.style.background='#e91e63';

    });

  });

  const fb=document.getElementById('mfb_'+id);

  fb.textContent=ok===answers.length?'✓ All correct!':'Some items need adjusting — highlighted in red.';

  fb.style.color=ok===answers.length?'#1b5e20':'#880e4f';

}


function renderQuiz(c){

  const key=currentMod+'_'+currentSlide;

  if(!quizState[key]) quizState[key]={q:0,score:0,done:false};

  const state=quizState[key];

  if(state.done) return`<div class="score-display"><div style="font-size:13px;color:#4a5568;margin-bottom:6px">Quiz complete</div><div class="score-big">${state.score}/${c.questions.length}</div><div style="font-size:12px;color:#4a5568;margin-top:4px">${state.score===c.questions.length?'Perfect score! 🎉':'Good effort — review the explanations above.'}</div></div>`;

  const qi=state.q, q=c.questions[qi];

  return`<div class="quiz-wrap">

    <div class="quiz-counter">Question ${qi+1} of ${c.questions.length}</div>

    <div class="quiz-q">${q.q}</div>

    ${q.scenario?`<div class="quiz-scenario">${q.scenario}</div>`:''}

    ${q.opts.map((o,i)=>`<button class="quiz-opt" onclick="answerQ(${i},${q.ans},'${key}',${c.questions.length},'${q.exp.replace(/'/g,"&#39;")}')">${String.fromCharCode(65+i)}) ${o}</button>`).join('')}

  </div>`;

}


function answerQ(chosen,correct,key,total,exp){

  document.querySelectorAll('.quiz-opt').forEach((o,i)=>{o.disabled=true;if(i===correct)o.classList.add('correct');else if(i===chosen)o.classList.add('wrong');});

  const fb=document.getElementById('qfb');

  fb.textContent=(chosen===correct?'✓ Correct! ':'✗ Incorrect. ')+exp.replace(/&#39;/g,"'");

  fb.className='quiz-feedback show '+(chosen===correct?'feedback-correct':'feedback-wrong');

  const state=quizState[key];

  if(chosen===correct) state.score++;

  setTimeout(async ()=>{

    state.q++;

    const mod=MODULES[currentMod], slide=mod.slides[currentSlide];

    if(state.q>=slide.content.questions.length) state.done=true;

    await saveCourseProgress();

    renderSlide();

  },2200);

}

async function deleteLearner(id){

}


async function exportCSV(){

  const learners = await loadLearners();

  const rows=[['Name','Email','Organisation','Role','Dzongkhag','Phone','Enrolled','Progress %','Status','Completed']];

  

  learners.forEach(l=>{

      new Date(l.enrolled_at || l.enrolledAt).toLocaleDateString(), p.pct||0,

  a.download='acc_learners_'+new Date().toISOString().slice(0,10)+'.csv';

  a.click();

}
